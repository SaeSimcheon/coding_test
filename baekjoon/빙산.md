- 크게 세가지 부분으로 나누어서 생각했음.
  - 굳이 count와 반영하는 부분을 나눈 이유는 같은 step에서 0이 된 자리의 영향을 받는 것을 막기 위해서
  - 빙산이 나누어졌는지 판단하는 부분
  - 개수를 주변의 0 개수를 count 하는 부분
  - count 개수를 반영하는 부분
- 위 세가지를 한 번에 처리할 방법이 떠오르지 않아서 따로 구현했음.
- python3로 하니까 시간 초과 나오네 (...) pypy는 통과함.
- 전부 녹을때까지 분리되지 않는 경우가 있었는데 이는 판별했을때 group 개수가 0이 되는 경우가 나오는 경우.

 

```python
import sys
from collections import deque
from tokenize import group
sys.stdin = open('input.txt','r')

R, C = map(int,sys.stdin.readline().split(' '))

position = deque([])
mat = []
for r in range(R):
    row = list(map(int,sys.stdin.readline().split()))
    mat.append(row)
    for idx,i in enumerate(row):
        if i != 0 :
            position.append((r,idx))

D = [(1,0),(-1,0),(0,1),(0,-1)]


# 그룹 개수를 세자



check = [[0]*C for _ in range(R)]




def mass():
    group =0 
    check = [[0]*C for _ in range(R)]
    flag = 0 
    for i in range(R) :
        for j in range(C) :
            if mat[i][j] != 0 and check[i][j] ==0:
                queue = deque([(i,j)])
                check[i][j] = 1
                while queue :
                    length = len(queue)

                    for _ in range(length) :
                        this = queue.popleft()
                        for d in D :
                            nx = this[0] + d[0]
                            ny = this[1] + d[1]

                            if mat[nx][ny] != 0 and check[nx][ny] == 0 :
                                check[nx][ny] = 1
                                queue.append((nx,ny))
                group +=1
            if group ==2 :
                flag =1
                break
        if flag == 1 :
            break
    return group



answer = 0
while position :
    count_list = deque([])
    for pos in position :
        count =0
        for d in D :
            nx = pos[0] + d[0]
            ny = pos[1] + d[1]
            if mat[nx][ny] == 0 :
                count +=1
        count_list.append(count)
    length = len(position)
    for _ in range(length) :
        this = position.popleft()
        count = count_list.popleft()
        if mat[this[0]][this[1]] - count <= 0 :
            mat[this[0]][this[1]] = 0
        else:
            mat[this[0]][this[1]] -= count
            position.append(this)

    oob = mass()
    if oob == 2 :
        answer +=1
        break
    elif oob == 0 :
        answer =0
    else: 
        answer +=1
print(answer)
```
