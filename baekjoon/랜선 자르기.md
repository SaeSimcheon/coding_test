```python
import sys

sys.stdin = open('input.txt','r')


K, N =map(int,sys.stdin.readline().split())


# N 개보다 많이 만드는 것도 포함 됨.
# 각 랜선을 얼마 단위로 잘라야 원하는 개수를 충족할 수 있는지
# 랜선의 길이가 최대가 되어야함.

# 일일이 나누어야 하나 ?
# 값중 가장 큰 것 기준 ?
# [하한, 상한] -> 정해지면 판별은 어떻게 ?

# 나눈 몫의 개수를 쭉 기록?

# 동일한 조건을 만족하는 최대

# 모두 순회하기 전에 기준 개수를 넘으면 후보로 넣어두고 
# 숫자를 키워서 다음을 탐색하면 됨.


from collections import deque

lan = list()

for _ in range(K):
    lan.append(int(sys.stdin.readline()))

lan.sort()

lan=deque(lan)

def checker(given,target):
    res = 0
    
    for i in lan :
        res += i // given
            
    return res


look = range(1,lan[-1]+1) # list로 만들어 두는 것보다 메모리가 훨씬 절약 됨.



a = 0 
b = len(look) -1

# 조건을 만족하는 상한을 찾을때까지 반복하면 됨.



while a <= b :
    point = (a+b)//2
    this = checker(look[point],N)
    
    if this >= N :
        a = point +1
    else :
        b = point -1



print(look[b])
```
