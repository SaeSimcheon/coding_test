- 에라토스테네스체 + 한쪽에서 시작하는 투포인터
- 연속적이지 않은 left 값을 관리를 위하여 deque를 사용하였음.

```python
import sys

sys.stdin = open('input.txt','r')

N=int(sys.stdin.readline())


# 에라토스 타네스 체

check = [0]*(N+1)
check[1] = 1
check[0] =1
summ=0
answer =0
from collections import deque
left =2
oob = deque([])
for i in range(2,N+1):
    if check[i] == 1 :
        continue
    summ += i
    oob.append(i)
    while summ > N :
        summ= summ - left
        oob.popleft()
        left = oob[0]
    if summ == N :
        answer +=1
    

    for j in range(i+i,N+1,i):
        check[j] = 1
print(answer)



```
