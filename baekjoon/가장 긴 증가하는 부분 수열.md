## lower bound를 이용한 풀이

```python
import sys

sys.stdin = open('input.txt','r')

N = int(sys.stdin.readline())

A = list(map(int,sys.stdin.readline().split()))




case = [A[0]]


def lower_bound(given):
    a = 0
    b = len(case)

    while a <b :
        point = (a+b) //2

        if case[point] < given :
            a = point +1
        else:
            b = point
    
    return a 



for i in A :
    if case[-1] < i :
        case.append(i)
    else:
        index = lower_bound(i)
        case[index] = i
    print(case)
print(case)
```
