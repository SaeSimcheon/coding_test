```pythoh
def solution(k, dungeons):
    global answer,check
    answer = 0
    check=[0]*len(dungeons)

    def dfs(remain) :
        global answer,check
        if sum([remain >= jj[0] for jj in dungeons]) ==0:
            can=sum(check)
            if can > answer :
                answer = can
            return 

        for idx,j in enumerate(dungeons) :
            if check[idx] == 0 and  remain >= j[0]:
                check[idx] =1
                dfs(remain - j[1])
                check[idx]=0


    dfs(k)
    return answer
 ```
