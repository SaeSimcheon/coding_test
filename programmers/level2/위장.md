def solution(clothes):

        seg,cat=i
        if cat in dictionary :
            dictionary[cat] +=1

            dictionary[cat] = 1
    count=[i for i in dictionary.values() ]
    length = len(count)
    answer = sum(count)
    for j in range(2,length+1):
        for jj in list(combinations(count,j)):
            can = 1

            answer+=can
