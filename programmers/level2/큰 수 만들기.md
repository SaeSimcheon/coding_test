```python
# 예를 달면 1924 k=2 
# 1을 탐색 k만큼인 9 2까지 보았을때 9보다 작으므로 1을 탈락시킴 -> 이때 k-=1
# 9에서 1만큼 확인했을때 9는 2보다 크므로 유지
# 2는 1만큼 확인했을때 4보다 작으므로 탈락.
# 현재 위치가 k == 0 이므로 종료


# k 개 안에 나보다 큰 경우가 있으면 나를 탈락시킴 만약 없다면 answer에 대입.
from collections import deque

def solution(number, k):
    number=deque(list(number))
    answer = ''
    while k >0 :
        
        length = k
        if length == len(number):
            break
        this = number.popleft()
        flag = 0
        for i in range(0,length):
            if number[i] > this :
                flag = 1
                k-=1
                break
            
        if flag == 0:
            answer += this
    
    if k != 0 : 
        return answer
    
    if len(number) !=0:
        return answer +''.join(number)
```

```python
# stack 사용

# k 감소
# 들어있는 수보다 뒤가 큰 경우에만 제거 나와 같거나 큰 녀석이 나올때까지 시행
# 길이가 0이 된 경우에도 시행 종료

from collections import deque

def solution(number, k):
    number=deque(list(number))
    answer = ''
    stack=deque([])
    
    while k >0 and number:
        this =number.popleft()
        length = len(stack)
        
        for j in range(length):
            comp = stack.pop()
            
            if comp >= this or k ==0:
                stack.append(comp)
                stack.append(this)
                break
            k-=1
        
        if not stack :
            stack.append(this)
    
    answer = ''.join(stack)
    
    if k > 0 : # k가 반복끝에 제외되지 않았음. 그 전에 number가 비어버림
        return answer[:-k]
    # '4321' case와 같이
    # 위를 시행하면 이런 결과가 나옴.
    # 실행한 결괏값 "4321"이(가) 기댓값 "43"와(과) 다릅니다. 
    
    if not number :
        return answer
    
    if number :
        answer += ''.join(number)
        return answer
```
